import java.util.*;
public class MyClass {
    public static void main(String args[]) {
        
        Scanner scn =new Scanner(System.in);
      int n= scn.nextInt(), b=scn.nextInt();
       
          int ans=0;
          int p=1;
          while(n>0){
              int rem=n%b;
              n=n/b;
              ans=ans+rem*p;
              p=p*10;
          }
      System.out.println("Decimal to base" +b+" = " + ans);
    }
}
